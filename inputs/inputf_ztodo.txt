struct Punto {
    int x;
    int y;
};

typedef int entero;

entero g[3];

int main() {
    struct Punto p;
    entero a[5];
    entero x;
    entero y;

    a[2] = 10;
    x = a[2];

    g[0] = x;
    g[1] = x + 5;
    g[2] = x * 2;

    p.x = g[1];
    p.y = g[2];     

    y = (p.x < p.y) ? p.x : p.y;

    printf(x);      
    printf(p.x);    
    printf(p.y);    
    printf(y);

    return 0;
}
